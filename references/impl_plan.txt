```jsx
// Social Proof widget — "X people viewed this doctor today"
export function SocialProofBadge({ doctorId }) {
  // Deterministic "random" based on doctorId for demo consistency
  const viewCount = (doctorId * 7 + 3) % 12 + 2  // 2-13
  return (
    <div className="flex items-center gap-1.5 text-xs text-trust-500">
      <span className="w-2 h-2 rounded-full bg-accent-400 animate-pulse inline-block" />

  )

// Scarcity badge — "Only X slots left!"
export function ScarcityBadge({ slotsLeft }) {
  if (slotsLeft > 3) return null

    <div className="flex items-center gap-1.5 text-xs font-semibold text-orange-600 bg-orange-50 border border-orange-200 px-2.5 py-1 rounded-full w-fit">
      <LucideIcons.Zap className="w-3.5 h-3.5" />
```

#### Create `src/components/ContextHelpTooltip.jsx`

A reusable tooltip that attaches to any form field:
```jsx
import { useState } from 'react'
import { Info } from 'lucide-react'

/*
 * Context-Sensitive Help — User Support Systems (end-sem HCI concept)
 * Task-specific: help is directly tied to what the user is currently doing
 * Unobtrusive: appears only on hover/click, never blocks workflow
 */
export default function ContextHelpTooltip({ text }) {
  const [visible, setVisible] = useState(false)
  return (
    <div className="relative inline-block">
      <button
        type="button"
        onMouseEnter={() => setVisible(true)}
        onMouseLeave={() => setVisible(false)}
        onClick={() => setVisible(v => !v)}
        className="text-trust-400 hover:text-primary-500 transition-colors ml-1"
        aria-label="Help"
      >
        <Info className="w-4 h-4" />
      </button>
      {visible && (
        <div className="absolute left-6 top-1/2 -translate-y-1/2 z-50 w-56 bg-trust-900 text-white text-xs rounded-xl p-3 shadow-xl leading-relaxed">
          {text}
          <div className="absolute right-full top-1/2 -translate-y-1/2 border-8 border-transparent border-r-trust-900" />
        </div>
      )}
    </div>
  )
}
```

#### Create `src/components/AssistantBubble.jsx`

```jsx
import { useState } from 'react'
import { HelpCircle, X, ChevronRight } from 'lucide-react'

/*
 * Non-intrusive Assistant — User Support Systems (end-sem HCI)
 * Contrast with Clippy: unobtrusive, under user control, relevant to context
 * Only shown on /booking route. Does not interrupt primary flow.
 */
export default function AssistantBubble() {
  const [open, setOpen] = useState(false)

  const steps = [
    { num: 1, text: 'Pick an available date and time slot.' },
    { num: 2, text: 'Enter your name and mobile number to confirm.' },
    { num: 3, text: 'Done! You\'ll see a confirmation with your booking summary.' },
  ]

  return (
    <div className="fixed bottom-6 right-6 z-50 flex flex-col items-end gap-3">
      {open && (
        <div className="bg-white rounded-2xl shadow-2xl border border-trust-200 w-72 p-5 animate-in">
          <div className="flex items-center justify-between mb-3">
            <p className="text-sm font-semibold text-trust-900">How booking works</p>
            <button onClick={() => setOpen(false)} className="text-trust-400 hover:text-trust-600">
              <X className="w-4 h-4" />
            </button>
          </div>
          <div className="space-y-3">
            {steps.map(step => (
              <div key={step.num} className="flex items-start gap-3">
                <span className="w-6 h-6 rounded-full bg-primary-100 text-primary-700 text-xs font-bold flex items-center justify-center shrink-0 mt-0.5">
                  {step.num}
                </span>
                <p className="text-sm text-trust-600 leading-snug">{step.text}</p>
              </div>
            ))}
          </div>
          <p className="text-xs text-trust-400 mt-4">Still stuck? <a href="/help" className="text-primary-600 underline">Visit Help →</a></p>
        </div>
      )}
      <button
        onClick={() => setOpen(o => !o)}
        className="w-12 h-12 rounded-full bg-primary-600 text-white shadow-lg hover:bg-primary-700 transition-all flex items-center justify-center"
        aria-label="Booking help"
      >
        {open ? <X className="w-5 h-5" /> : <HelpCircle className="w-5 h-5" />}
      </button>
    </div>
  )
}
```

```jsx
import { useNavigate } from 'react-router-dom'
import { ChevronRight } from 'lucide-react'

/*
 * Navigation Design Guideline: breadcrumbs give users their location in hierarchy
 * Synthesizability (Lecture 4): users can assess what they've done and where they are

export default function Breadcrumb({ items }) {
  /* items = [{ label: 'Home', path: '/' }, { label: 'Dr. Name' }] */

              className="hover:text-primary-600 transition-colors"
            >
              {item.label}
            </button>
          ) : (

          )}
        </span>
      ))}
```
